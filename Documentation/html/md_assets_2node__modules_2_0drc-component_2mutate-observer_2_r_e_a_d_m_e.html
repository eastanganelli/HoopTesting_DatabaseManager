Install

Usage

import React from 'react';
import MutateObserver from './src';
const onMutate = (mutations: MutationRecord[], observer: MutationObserver) => {
console.log(mutation);
console.log(observer);
};
const Demo: React.FC = () => {
return (
<MutateObserver onMutate={onMutate}>
<div>test</div>
</MutateObserver>
);
};
export default Demo;
